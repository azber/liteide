LiteEnv Plugin

LiteIDE environment setup plugin

Select Environment

Select current use environment, default system

  • Windows - win64 or win32
  • Linux - linux64 or linux32
  • MacOSX - darwin64 or darwin32

Configure Environment

Menu: View -> Options -> LiteEnv or Welcome: Options -> LiteEnv

Double clicked *.env file to editor

Example win32.env

#win32 environment

GOROOT=c:\go
GOARCH=386
GOOS=windows

PATH=%GOROOT%\bin;%PATH%

LITEIDE_GDB=gdb
LITEIDE_MAKE=mingw32-make
LITEIDE_TERM=%COMSPEC%
LITEIDE_TERMARGS=
LITEIDE_EXEC=%COMSPEC%
LITEIDE_EXECOPT=/k

Example linux32.env

#linux32 environment

GOROOT=$HOME/go
GOARCH=386
GOOS=linux

PATH=$GOROOT/bin:$PATH

LITEIDE_GDB=gdb
LITEIDE_MAKE=make
LITEIDE_TERM=/usr/bin/gnome-terminal
LITEIDE_TERMARGS=
LITEIDE_EXEC=/usr/bin/xterm
LITEIDE_EXECOPT=-e

LiteBuild Plugin

LiteIDE build system, build project or file, if file belong a project

Configure Build

Menu: View -> Options -> LiteBuild or Welcome: Options -> LiteBuild

Double clicked *.xml file to editor

Example linux gosrc.xml

<?xml version="1.0" encoding="UTF-8"?>
<mime-info xmlns='http://www.freedesktop.org/standards/shared-mime-info'>
<mime-type type="text/x-gosrc" id="gosrc">
<lookup mime-type="text/x-gomake" file="Makefile" top="1"/>
<lookup mime-type="text/x-gopro" file="*.e64" top="1"/>
<action id="Build" img="go/build.png" key="F7" cmd="${LITEAPPDIR}/gopromake" args="-gofiles ${EDITORNAME} -o ${TARGETNAME}" save="editor" output="true" codec="utf-8" regex="([\w\d:_\\/\.]+):(\d+)" />
<action id="Clean" img="go/clean.png" cmd="${LITEAPPDIR}/gopromake" args="-gofiles ${EDITORNAME} -o ${TARGETNAME} -clean=obj" output="true" codec="utf-8"/>
<action id="CleanAll" img="go/cleanall.png" cmd="${LITEAPPDIR}/gopromake" args="-gofiles ${EDITORNAME} -o ${TARGETNAME} -clean=all" output="true" codec="utf-8"/>
<action id="Exec" img="go/exec.png" key="F5" cmd="${LITEAPPDIR}/goexec" args="${TARGETNAME}" output="false"/>
<action id="Run" img="go/run.png" key="CTRL+F5" cmd="${TARGETPATH}" args="" output="true" codec="utf-8"/>
<action id="BuildAndRun" img="go/buildandrun.png" key="CTRL+F7" task="Build;Run"/>
</mime-type>
</mime-info>
  • lookup mime-type: if no project, load example.go first lookup Makefile or *.e64 to use build toolbar
  • action task: order execution defined action id list

Example linux gomake.xml

<?xml version="1.0" encoding="UTF-8"?>
<mime-info xmlns='http://www.freedesktop.org/standards/shared-mime-info'>
<mime-type type="text/x-gomake" id="gomake">
<action id="Build" img="go/build.png" key="F7" cmd="gomake" args="${PROJECTNAME} all" save="project" output="true" codec="" regex="([\w\d:_\\/\.]+):(\d+)" />
<action id="Rebuild" img="go/rebuild.png" key="CTRL+F7" cmd="gomake" args="-B ${PROJECTNAME} all" save="project" output="true" codec="" regex="([\w\d:_\\/\.]+):(\d+)" />
<action id="Clean" img="go/clean.png" cmd="gomake" args="${PROJECTNAME} clean" save="project" output="true"/>
<action id="Nuke" img="go/nuke.png" cmd="gomake" args="${PROJECTNAME} nuke" save="project" output="true"/>
<action id="Install" img="go/install.png" cmd="gomake" args="${PROJECTNAME} install" save="project" output="true"/>
<action id="Test" img="go/test.png" key = "ALT+F5" cmd="gotest" args="${PROJECTNAME}" save="project" output="true" codec="utf-8"/>
<action id="Exec" img="go/exec.png" key="F5" cmd="/usr/bin/xterm" args="-e ${LITEAPPDIR}/goexec  ${TARGETNAME}" output="false"/>
<action id="Run" img="go/run.png" key="CTRL+F5" cmd="${TARGETPATH}" args="" output="true" codec="utf-8"/>
<action id="BuildAndRun" img="go/buildandrun.png" key="CTRL+F7" task="Build;Run"/>
</mime-type>
</mime-info>

LiteDebug Plugin

LiteIDE debug plugin

Gdb version >= 7.1

Golang debug request gdb version 7.1 or high

Source path

The file or project source path cannot include outsite space or latin1 char

Environment LITEIDE_GDB

example LiteEnv -> win64.env

PATH=c:/go-w64/bin;%PATH%
GOROOT=c:/go-w64
GOBIN=c:/go-w64/bin
GOARCH=amd64
GOOS=windows
LITEIDE_GDB=gdb64

WIN32 and WIN64

WIN32 request gdb.exe,WIN64 request gdb64.exe.
